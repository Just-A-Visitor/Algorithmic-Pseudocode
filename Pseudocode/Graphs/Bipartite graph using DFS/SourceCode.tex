\begin{algorithm}

  \caption{Check if a connected graph is bipartite using BFS}
  
  \algorithmicrequire \ A connected graph with V vertices\\
  \algorithmicensure \ $true$ if the graph is bipartite, $false$ otherwise
  
  \begin{algorithmic}[1]
    
    \Statex
    
    \Function{$isBipartite$}{}
    
    
        \State $Queue\ queue$ \Comment{Used for the BFS}
        \For {each $vertex$}
            \State $color[i] \gets 0$
            \State $visited[i] \gets false$
        \EndFor
        \State $queue.push(0)$
        \State $visited[0] \gets true$
        
        \Statex
        \While{$queue\ is\ \textbf{not\ empty}$} \Comment{BFS}
            \State $current \gets queue.front$
            \State $queue.pop$
            \For{each vertex $v$ adjacent to $current$} \Comment{Where $v$ is the vertex's index}
                \If{$\textbf{not}\ visited[v]$}
                    \State $queue.push(v)$
                    \State $color[v] \gets color[current] \oplus 1$ \Comment{Assign $v$ in a different set}
                    \State $visited[v] \gets true$
                \Else
                    \If{$color[v]=color[current]$}
                        \State \Return $false$
                    \EndIf
                \EndIf
            \EndFor
        \EndWhile
        \Statex
        \State \Return $true$
    \EndFunction
  \end{algorithmic}
  
\end{algorithm}
