\begin{algorithm}

  \caption{Check if a connected graph is bipartite using BFS}
  
  \algorithmicrequire \ A connected graph with V vertices\\
  \algorithmicensure \ $true$ if the graph is bipartite, $false$ otherwise
  
  \begin{algorithmic}[1]
    
    \Statex
    
    \Function{$isBipartite$}{}
    
        \LeftComment{Initialization}
        \State $Queue\ q$ \Comment{Used for the BFS}
        \For {$i = 1 : V$} \Comment{Where each $i$ represents a vertex}
            \State $color[i] \gets 0$
            \State $visited[i] \gets false$
        \EndFor
        \State $q.push(0)$ \Comment{Index of the first node to explore}
        \State $visited[0] \gets true$
        
        \Statex
        \LeftComment{BFS}
        \While{$q\ is\ \textbf{not\ empty}$}
            \State $current \gets q.pop()$
            \For{each vertex $v$ adjacent to $current$} \Comment{Where $v$ is the vertex's index}
                \If{$\textbf{not}\ visited[v]$}
                    \State $q.push(v)$ \Comment{To later check its adjacent vertices}
                    \State $color[v] \gets color[curent] \oplus 1$
                    \State $visited[v] \gets true$
                \Else
                    \If{$color[v]=color[current]$}
                        \State \Return $false$ \Comment{Two adjacent vertices must have different colors}
                    \EndIf
                \EndIf
            \EndFor
        \EndWhile
        \State \Return $true$
    \EndFunction
  \end{algorithmic}
  
\end{algorithm}
