            \State $min\_heap.push(a[i])$
        \EndFor
        
        \State $cost \gets 0$
        \While{$min\_heap.size > 1$}
            \State $min\_element \gets min\_heap.top$, $min\_heap.pop$
            \State $second\_min\_element \gets min\_heap.top$, $min\_heap.pop$
            
            \State $cost += min\_element + second\_element$

        \EndWhile
    \EndFunction
