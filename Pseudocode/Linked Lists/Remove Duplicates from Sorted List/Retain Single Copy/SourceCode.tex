\begin{algorithm}

  \caption{Remove all duplicates from a sorted list }
  
  \begin{algorithmic}[1]
    \Ensure You should retain one copy of the duplicate elements
    \Statex
    
    \Function{Remove\_Duplicates}{$head$}
    
        \If{List is of length $0$ \textit{or} $1$}
            \State \Return{$head$}
        \EndIf
        
        \Statex
        \LeftComment{Recursively remove the duplicates to the right and connect the two lists}
        \Statex
        
        \State $clean\_sub\_list \gets Remove\_Duplicates(head.next)$
        \State $head.next \gets clean\_sub\_list$
        \Statex
        
        \If{$head.data$ $==$ $head.next.data$}
            \State \Return {$clean\_sub\_list$}
        \Else
            \State \Return {$head$}
            
        \EndIf
    \EndFunction
  \end{algorithmic}
  
\end{algorithm}
