    \begin{algorithm}
    
      \caption{Stock Span}
      
      \begin{algorithmic}[1]
        \Ensure One Based Indexing
        \Statex
        \Function {$Calculate\_Stock\_Span$}{$price$}
            \LeftComment{$span[i]$ denotes the span of the price of stock at the $i-th$ day}
            
            \Statex
            \State $len \gets price.length$
            \For{$ i = 1 \To len$}
                \While{$stack$ \IsNotEmpty \ \And \ $price[stack.top] \leq price[i]$}
                    \State $stack.pop$
                \EndWhile
                
                \Statex
                \If{$stack$ \IsNotEmpty}
                    \State $span[i] \gets (i - stack.top)$
                \Else 
                    \State $span[i] \gets i$ \Comment{One Based Indexing}
                    \Statex
                \EndIf
                
                \State{$stack.push(i)$}
                
            \EndFor
            \Statex
            \State \Return{$span$}
        \EndFunction
      \end{algorithmic}
      
    \end{algorithm}
