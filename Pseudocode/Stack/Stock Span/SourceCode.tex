    \begin{algorithm}
    
      \caption{Stock Span}
      
      \begin{algorithmic}[1]
        \Ensure One Based Indexing
        \Statex
        \Function {$Calculate\_Stock\_Span$}{$prices,len$}
            \LeftComment{$span[i]$ denotes the span of the price of stock at the $i-th$ day}
            
            \Statex
            \For{$ i = 1 : len$}
                \Statex
                \While{$stack \IsNotEmpty$}
                    \If{$price$[$stack.top$] $<=$ $price$[$i$]}
                        \State{$stack.pop$}
                    \Else 
                        \State{\Break} Loop
                    \EndIf
                \EndWhile
                
                \Statex
                \If{$stack \IsNotEmpty $}
                    \State $span[i] \gets (i - stack.top)$
                \Else 
                    \State $span[i] \gets (i+1)$
                \EndIf
                
                \State{$stack.push(i)$}
                
            \EndFor
            \Statex
            \State \Return{$span$}
        \EndFunction
      \end{algorithmic}
      
    \end{algorithm}
