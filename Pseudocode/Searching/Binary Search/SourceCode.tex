\begin{algorithm}

  \caption{Searching an element in a sorted array}
  
\begin{algorithmic}[1]
	\Statex
	\Procedure{BinarySearch}{$A$,$l$,$r$,$x$}
	\If {$l geq r$}
		\State $m \gets \lfloor l + (r-1) / 2) \rfloor$
		\If {$A[m] == x$}
			\State \Return $m$
		\EndIf
		\If {$A[m] > x$}
			\State \Return $BinarySearch(A,l,m-1,x)$
		\Else
			\State \Return $BinarySearch(A,m+1,r,x)$
		\EndIf
	\EndIf
\EndProcedure
\end{algorithmic}

\end{algorithm}
