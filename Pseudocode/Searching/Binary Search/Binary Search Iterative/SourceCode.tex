\begin{algorithm}

  \caption{Searching an element in a sorted array} 
\begin{algorithmic}[1]
	\Statex
	\Procedure{BinarySearch}{$A$, $l$, $r$, $x$}
	\While{$l \leq r$}
    	\State $m \gets (l + r) \, / \, 2$
    	\If{$A[m] = x$}
        	\State \Return $m$
    	\EndIf 
    	
    	\If{$x < A[m]$}
    	    \State $r = m - 1$
    	\Else
        	\State $l = m + 1$ 
    	\EndIf
	\EndWhile
    \\
    \State \Return $-1$
\EndProcedure
\end{algorithmic}

\end{algorithm}
