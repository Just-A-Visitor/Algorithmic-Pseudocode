\begin{algorithm}

  \caption{Function to find the shortest subarray sorting which sorts the entire array}
  \begin{algorithmic}[1]
    \Ensure One Based Indexing for the array
    \Statex
    \Function{$Find\_Sub\_Array$}{$arr$}
        \LeftComment{$min\_right[i]$ denotes the maximum element to the right of i}
        \LeftComment{$min\_left[i]$ denotes the maximum element to the left of i}
        \Statex
        
        \State $n \gets arr.length$
        \If{$n = 0$ \textbf{or} $n = 1$}
            \State \Return{0}
        \EndIf
        
        \Statex
        \State $min\_right[n - 1] \gets arr[n - 1]$
        \For{$ i = n$ - 1 : -1}
            \State $min\_right[i] \gets min(arr[i], min\_right[i + 1])$
        \EndFor
        
        \Statex
        \State $max\_left[0] \gets arr[0]$
        \For{$ i = 1 : n$}
            \State $max\_left[i] \gets min(arr[i], max\_left[i - 1])$
        \EndFor
        
        \Statex
        \State $left\_index \gets -1$
        \For{$ i = 0 : n$}
            \If{$arr[i] \neq min\_right[i]$}
                \State $left\_index \gets i$
                \State{\Break} Loop
            \EndIf
        \EndFor
        
        \Statex
        \State $right\_index \gets -1$
        \For{$ i = n$ - 1$ : -1$}
            \If{$arr[i] \neq max\_left[i]$}
                \State $right\_index \gets i$
                \State{\Break} Loop
            \EndIf
        \EndFor
        
        \Statex
        \If{$left\_index = -1$ \textbf{or} $right\_index = -1$}
            \State \Return{0}
        \Else
            \State \Return{$right\_index - left\_index$ + 1 }
        \EndIf
    \EndFunction
  \end{algorithmic}
  
\end{algorithm}
