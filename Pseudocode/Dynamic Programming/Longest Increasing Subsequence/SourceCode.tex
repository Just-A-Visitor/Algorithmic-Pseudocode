\begin{algorithm}

  \caption{Find the length of the longest increasing subsequence}
  
  \begin{algorithmic}[1]
    \Ensure One Based Indexing
    \Statex
    \Function{LIS}{$Arr, len$}
    
        \LeftComment{$lis[i]$ denotes the length of the LIS which starts at the $i-th$ index}
        \Statex
        
        \For{$ i = len : 1$}
            \State $lis[i] \gets 1$
        \EndFor
        
        \For{$ i = len : 1$}
            \For{$ j = i + 1 : len$}
                \If{$a[j] > a[i]$}
                    \State $lis[i] \gets max(lis[i], 1 + lis[j])$
                \EndIf
            \EndFor
        \EndFor
        
        \State \Return{Maximum element of $LIS$ array}
    \EndFunction
  \end{algorithmic}
  
\end{algorithm}
