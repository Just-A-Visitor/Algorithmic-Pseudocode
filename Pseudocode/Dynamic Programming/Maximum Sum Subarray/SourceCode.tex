\begin{algorithm}

  \caption{Find the length of the subarray with the maximum sum}
  
  \begin{algorithmic}[1]
    \Ensure One Based Indexing
    \Statex
    
    \Function{Kadane}{$Arr, len$}
    
        \LeftComment{$start[i]$ denotes the length of the maximum sum subarray which starts at the $i-th$ index}
        \Statex
        
        \State $start[len] \gets a[len]$
        \For{$ i = len-1 : 1$}
            \State $start[i] \gets max(a[i] + start[i+1], a[i])$
        \EndFor
        
        \State \Return{Maximum element of $Start$ array}
    \EndFunction
  \end{algorithmic}
  
\end{algorithm}
