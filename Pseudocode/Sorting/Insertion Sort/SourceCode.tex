\begin{algorithm}

  \caption{Sort an array one by one}

  \begin{algorithmic}[1]
    \Ensure One Based Indexing for the Array
    \Require A non empty array
    \Statex

    \Function{$Insertion\_Sort$}{$arr$}
    \LeftComment $arr$ is the array to be sorted.
    
        \Statex

        \For{$j = 2 \To length(arr)$}
            \State $key \gets arr[j]$
            
            \Statex
            
            \State $i \gets j-1$
            \While{$i >= 1$ \And $arr[i] > key$}
                \State $arr[i+1] \gets arr[i]$
                \State $i \gets i-1$
            \EndWhile
            
            \Statex
            \State $arr[i+1] \gets key$
            
        \EndFor
        
        \Statex
        \State \Return{$arr$} 

    \EndFunction
  \end{algorithmic}

\end{algorithm}
