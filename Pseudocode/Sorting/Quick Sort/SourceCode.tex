\begin{algorithm}

  \caption{Sort an array using Divide and Conquer approach}
  
  \begin{algorithmic}[1]
    \Statex

    \Procedure{Partition}{$arr,\ low,\ high$}
        \State $pivot = arr[high]$
        \State $i \gets low - 1$
        \For {$j \gets low \To high - 1$}
            \If{$arr[j] \leq pivot$}
                \State $i \gets i + 1$
                \State $swap(arr[i],\ arr[j])$
            \EndIf
        \EndFor
        \Statex
        \State $swap(arr[i + 1],\ arr[high])$
        \State \Return $i + 1$
    \EndProcedure
    \Statex
    \Function{Quick\_Sort}{$arr,\ low,\ high$}
        \If{$low < high$}
            \State $pi\ \gets\ $\textbf{\textit{Partition}}$(arr,\ low,\ high)$
            \State \textbf{Quick\_Sort}$(arr,\ low,\ pi\ -\ 1)$
            \State \textbf{Quick\_Sort}$(arr,\ pi\ +\ 1,\ high)$
        \EndIf
    \EndFunction
  \end{algorithmic}
  
\end{algorithm}
