  \caption{Sort an array in non-descending order}
  
\begin{algorithmic}[1]
	\Statex
	\Function{SelectionSort}{$A$}
	\For {$i=1,\dots,n-1$}
		\State $m \gets i$
		\For {$j=i+1,\dots,n$}
			\If {$M[j]<M[m]$}
			\State $m \gets j$
			\EndIf
		\EndFor
		\State $swap(M[i],M[m])$ 	
	\EndFor
\EndFunction
\end{algorithmic}

\end{algorithm}
