\usepackage{algorithm}

\usepackage[noend]{algpseudocode}

% Define Left Justified Comments
\algnewcommand{\LeftComment}[1]{\Statex \(\triangleright\) #1}

% Remove the Numbering of the Algorithm
\usepackage{caption}
\DeclareCaptionLabelFormat{algnonumber}{Algorithm}
\captionsetup[algorithm]{labelformat = algnonumber}

\newcommand{\IsNotEmpty}{\textbf{ is not empty, }}
\newcommand{\Break}{\textbf{break}}
\newcommand{\Print}{\textbf{Print}}

\begin{document}

    \begin{algorithm}
    
      \caption{Find the first non-repeating character when each character of the stream arrives}
      
      \begin{algorithmic}[1]
        \Statex
        \Function {$First\_Non\_Repeating\_Char$}{$str$}
            \LeftComment{$frequency$ is a hashmap which stores the frequency of the characters}
            \LeftComment{$queue$ is a queue of characters}
            \State{Scan $str$ from left to right and,}
            \For{each character $char$}
                \State{$queue$.push($char$)}
                \State{Increment $frequency$[$char$] by 1}
                \While{$queue$ \IsNotEmpty}
                    \If{$frequency$[$queue.front$] $>$ 1}
                        \State{$queue.$pop()}
                    \Else 
                        \State{\Break}
                    \EndIf
                \EndWhile
                \If{$queue$ \IsNotEmpty}
                    \State{\Print($queue.front$)}
                \Else 
                    \State{\Print($null)$}
                \EndIf
                
            \EndFor
            \Statex
            \State \Return{}
        \EndFunction
      \end{algorithmic}
      
    \end{algorithm}
