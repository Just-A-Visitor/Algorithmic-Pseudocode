    \begin{algorithm}
    
      \caption{Find the first non-repeating character when each character of the stream arrives}
      
      \begin{algorithmic}[1]
        \Statex
        \Function {$First\_Non\_Repeating\_Char$}{$stream$}
            \LeftComment{$freq$ is a hashmap which stores the frequency of the characters}
            
            \Statex
            \For{each character $char$ \textbf{in} $stream$}
                \State{$queue.push(char)$}
                \State $freq[char] \gets freq[char] + 1$ \Comment{Assumes default frequency as 0}
                
                \Statex
                \While{$queue \IsNotEmpty$}
                    \If{$freq$[$queue.front$] $>$ 1}
                        \State{$queue.pop$}
                    \Else 
                        \State{\Break} Loop
                    \EndIf
                \EndWhile
                
                \Statex
                \If{$queue \IsNotEmpty $}
                    \State{\Print($queue.front$)}
                \Else 
                    \State{\Print(No unique character)}
                \EndIf
                
            \EndFor
            \Statex
            \State \Return{}
        \EndFunction
      \end{algorithmic}
      
    \end{algorithm}
