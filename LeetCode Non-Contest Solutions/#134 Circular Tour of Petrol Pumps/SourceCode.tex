\begin{algorithm}

    \caption{Circular Tour of Petrol Pumps}
    
    
    \begin{algorithmic}[1]
    \Statex
    \Function{Can\_Complete\_Circuit}{$gas,cost$}
        \LeftComment $current\_gas$ contains the current gas quantity
        \LeftComment $start$ is the start position
        \LeftComment $total\_gas$ contains the total quantity of gas
        
        \Statex
        \State $current\_gas \gets 0$
        \State $start \gets 0$
        \State $total\_gas \gets 0$
        
        \For{$j = 2 \To length(arr)$}
            \State $dif \gets gas[j] - cost[j]$
            \State $total\_gas \gets total\_gas + dif$
            \State $current\_gas \gets current\_gas + dif$
            
            \If {$current\_gas < 0$}
                \State $start \gets j + 1$
                \State $current\_gas \gets 0$
            
            \EndIf
        
        \EndFor
        
        \If{$total\_gas \geq 0$}
            \State \Return {$start$}
        
        \EndIf
        
        \State \Return{$-1$}
    \EndFunction
\end{algorithmic}
\end{algorithm}
